```
from collections import defaultdict

s = [('red', 1), ('blue', 2), ('red', 3), ('blue', 4), ('red', 1), ('blue', 4)]
d = defaultdict(set)
print(d)
for k, v in s:
	print("k= %s" %k)
	print("v= %s" %v)
	print("1d[k] = %s" %d[k])
	d[k].add(v) #set()创建一个空集合，add方法是集合的方法，将元素v放到集合中
	print("2d[k] = %s" % d[k])

print(d)
```
